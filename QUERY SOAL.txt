SELECT POWER_UNIT.POWER_UNIT_NUM, POWER_UNIT.DESCRIPTION,CORPORATION.CORPORATION_NAME,LOCATION.LOCATION_NAME,POWER_UNIT_TYPE.POWER_UNIT_TYPE_XID
FROM POWER_UNIT
JOIN CORPORATION ON POWER_UNIT.ID_CORPORATION = CORPORATION.ID_CORPORATION
JOIN LOCATION ON POWER_UNIT.ID_LOCATION = LOCATION.ID_LOCATION
JOIN POWER_UNIT_TYPE ON POWER_UNIT.ID_POWER_UNIT_TYPE = POWER_UNIT_TYPE.ID_POWER_UNIT_TYPE
